 21def solve_eqn(aaa, bbb, base_vec, crf_centre):
 22    """djb to document
 23
 24    Args:
 25        aaa (_type_): _description_
 26        bbb (_type_): _description_
 27        base_vec (_type_): _description_
 28        crf_centre (_type_): _description_
 29
 30    Returns:
 31        _type_: _description_
 32    """
 33    crf_point = np.zeros(3)
 34
 35    d_xplus = 2.0 * bbb * base_vec[0] / np.power(base_vec[2], 2.0)
 36
 37    d_befosqrt = 4.0 * np.power(bbb, 2.0) * np.power(base_vec[0], 2.0) / np.power(base_vec[2], 4.0)
 38    d_befosqrt -= (
 39        4.0
 40        * (np.power(base_vec[0], 2.0) / np.power(base_vec[2], 2.0) + 1.0)
 41        * (np.power(bbb, 2.0) / np.power(base_vec[2], 2.0) - aaa)
 42    )
 43
 44    d_xplus += np.sqrt(d_befosqrt)
 45    d_xplus /= 2.0 * (np.power(base_vec[0], 2.0) / np.power(base_vec[2], 2.0) + 1.0)
 46
 47    d_xxplus = d_xplus + crf_centre[0]
 48
 49    d_zplus = (bbb - base_vec[0] * d_xplus) / base_vec[2]
 50
 51    d_zzplus = d_zplus + crf_centre[2]
 52
 53    d_xminus = 2.0 * bbb * base_vec[0] / np.power(base_vec[2], 2.0)
 54    d_xminus -= np.sqrt(d_befosqrt)
 55    d_xminus /= 2.0 * (np.power(base_vec[0], 2.0) / np.power(base_vec[2], 2.0) + 1.0)
 56
 57    d_xxminus = d_xminus + crf_centre[0]
 58
 59    d_zminus = (bbb - base_vec[0] * d_xminus) / base_vec[2]
 60
 61    d_zzminus = d_zminus + crf_centre[2]
 62
 63    if d_xxplus > d_xxminus:
 64        crf_point[0] = d_xxplus
 65        crf_point[2] = d_zzplus
 66
 67    if d_xxminus > d_xxplus:
 68        crf_point[0] = d_xxminus
 69        crf_point[2] = d_zzminus
 70
 71    return crf_point

 74def get_crf_in_efc(lon, lat, alt, vel_vec):
 75    """djb to document
 76
 77    Args:
 78        lon (_type_): _description_
 79        lat (_type_): _description_
 80        alt (_type_): _description_
 81        vel_vec (_type_): _description_
 82
 83    Returns:
 84        _type_: _description_
 85    """
 86    crf_axis = np.zeros((3, 3))
 87    efc_cog = np.zeros(3)
 88    nad = np.zeros(3)
 89    nad[0], nad[1], nad[2] = llh_to_ecef_pyproj(lat, lon, 0.0)
 90    log.debug("NADIR LLH %f %f %f", lat, lon, 0.0)
 91    log.debug("NADIR EFC %f %f %f len %f", nad[0], nad[1], nad[2], np.linalg.norm(nad))
 92
 93    efc_cog[0], efc_cog[1], efc_cog[2] = llh_to_ecef_pyproj(lat, lon, alt)
 94
 95    sat_nad_vec = nad - efc_cog
 96    log.debug(
 97        "SAT NAD VEC EFC %f %f %f len %f",
 98        sat_nad_vec[0],
 99        sat_nad_vec[1],
100        sat_nad_vec[2],
101        np.linalg.norm(sat_nad_vec),
102    )
103
104    ad_crf_axis1 = sat_nad_vec / np.linalg.norm(sat_nad_vec)
105    ad_efc_nv = vel_vec / np.linalg.norm(vel_vec)
106
107    scal_prod = np.dot(ad_crf_axis1, ad_efc_nv)
108    ad_temp_vect = np.zeros(3)
109    ad_temp_vect[0] = ad_efc_nv[0] - ad_crf_axis1[0] * scal_prod
110    ad_temp_vect[1] = ad_efc_nv[1] - ad_crf_axis1[1] * scal_prod
111    ad_temp_vect[2] = ad_efc_nv[2] - ad_crf_axis1[2] * scal_prod
112
113    ad_crf_axis2 = ad_temp_vect / np.linalg.norm(ad_temp_vect)
114    ad_crf_axis3 = np.cross(ad_crf_axis2, ad_crf_axis1)
115
116    # crf_axis[0, :] = ad_crf_axis1
117    # crf_axis[1, :] = ad_crf_axis2
118    # crf_axis[2, :] = ad_crf_axis3
119    # This fixed a problem, but I don't know if I'm abusing vector orientation here
120    crf_axis[:, 0] = ad_crf_axis1
121    crf_axis[:, 1] = ad_crf_axis2
122    crf_axis[:, 2] = ad_crf_axis3
123
124    return crf_axis, efc_cog

127def rotation_matrix(crf_axis):
128    """djb to document
129
130    Args:
131        crf_axis (_type_): _description_
132
133    Returns:
134        _type_: _description_
135    """
136    rot = np.zeros((3, 3))
137    rot[0, 0] = np.dot([1, 0, 0], crf_axis[:, 0])
138    rot[0, 1] = np.dot([1, 0, 0], crf_axis[:, 1])
139    rot[0, 2] = np.dot([1, 0, 0], crf_axis[:, 2])
140    rot[1, 0] = np.dot([0, 1, 0], crf_axis[:, 0])
141    rot[1, 1] = np.dot([0, 1, 0], crf_axis[:, 1])
142    rot[1, 2] = np.dot([0, 1, 0], crf_axis[:, 2])
143    rot[2, 0] = np.dot([0, 0, 1], crf_axis[:, 0])
144    rot[2, 1] = np.dot([0, 0, 1], crf_axis[:, 1])
145    rot[2, 2] = np.dot([0, 0, 1], crf_axis[:, 2])
146    return rot

149def angle_to_poca(angle, lat, lon, alt, cor_range, vel_vec, base_vec):
150    """djb to document
151
152    Args:
153        angle (_type_): _description_
154        lat (_type_): _description_
155        lon (_type_): _description_
156        alt (_type_): _description_
157        cor_range (_type_): _description_
158        vel_vec (_type_): _description_
159        base_vec (_type_): _description_
160
161    Returns:
162        _type_: _description_
163    """
164    #    base_vec[0] = 0.0
165    #    base_vec[1] = 0.0
166    #    base_vec[2] = -1.0
167    log.debug("cor_range %f", cor_range)
168    log.debug("alt-range %f", alt - cor_range)
169    log.debug("vel_vec %s", str(vel_vec))
170    log.debug("base_vec %s", str(base_vec))
171
172    crf_centre = base_vec * (cor_range * np.sin(angle))
173    log.debug("crf_centre %s", str(crf_centre))
174
175    radius = cor_range * np.cos(angle)
176    log.debug("radius %f", radius)
177
178    aaa = np.power(radius, 2) - np.power(crf_centre[1], 2)
179    bbb = base_vec[1] * crf_centre[1]
180
181    crf_point = solve_eqn(aaa, bbb, base_vec, crf_centre)
182    log.debug("crf_point %s", str(crf_point))
183
184    try:
185        crf_axis, efc_cog = get_crf_in_efc(lon, lat, alt, vel_vec)
186
187    except ValueError as exc:
188        log.error("Geolocation failed, Floating point exception: %s", exc)
189        return np.nan, np.nan, np.nan
190    rot = rotation_matrix(crf_axis)
191
192    efc_vec = np.inner(rot, crf_point)
193    log.debug("efc_vec %s len %f", str(efc_vec), np.linalg.norm(efc_vec))
194    log.debug("efc_cog %s len %f", str(efc_cog), np.linalg.norm(efc_cog))
195    efc_point = efc_vec + efc_cog
196    log.debug("efc_point %s len %f", str(efc_point), np.linalg.norm(efc_point))
197    lat_poca, lon_poca, elev_poca = ecef_to_llh_pyproj(efc_point[0], efc_point[1], efc_point[2])
198    log.debug("SAT lat=%f lon=%f h=%f", lat, lon, alt)
199    log.debug("POCA lat=%f lon=%f h=%f", lat_poca, lon_poca, elev_poca)
200    if lon_poca > 180.0:
201        lon_poca -= 360.0
202
203    return lat_poca, lon_poca, elev_poca

206def phase_to_angle(
207    phase,
208    wavelength=0.022084,
209    baseline=1.1676,
210    inferred_angle_cal_mult=1.02775,
211    inferred_angle_cal_add=0.0,
212):
213    """djb to document
214
215    Args:
216        phase (_type_): _description_
217        wavelength (float, optional): _description_. Defaults to 0.022084.
218        baseline (float, optional): _description_. Defaults to 1.1676.
219        inferred_angle_cal_mult (float, optional): _description_. Defaults to 1.02775.
220        inferred_angle_cal_add (float, optional): _description_. Defaults to 0.0.
221
222    Returns:
223        _type_: _description_
224    """
225    angle = phase * wavelength
226    angle = -1.0 * angle / (2 * np.pi * baseline)
227
228    angle = (inferred_angle_cal_mult * angle) + inferred_angle_cal_add
229    return angle

270    final_lon_20_ku = np.zeros(nrec)

272    angle_20_ku = np.zeros(nrec)
273
274    lat_initial_20_ku = np.zeros(nrec)
275    lon_initial_20_ku = np.zeros(nrec)

277    lon_unwrap_20_ku = np.zeros(nrec)
278    dem_unwrap_20_ku = np.zeros(nrec)

280    delta_unwrap_h = np.zeros(nrec)
281    height_unwrap_20_ku = np.zeros(nrec)

284    if config_fitter == 1:
285        fitter = sarin_phase.phase_fit_lsq
286    elif config_fitter == 2:
287        fitter = sarin_phase.phase_fit_cuf
288    else:

296    bad_2 = 0
297    bad_3 = 0

302    )

307    # ------------------------------------------------------------------------------
308

310        log.debug("processing record %d", i)

312        if complete >= log_completed:

314            log.info("Completed %d%%", int(complete))
315        try:
316            # Check if inputs are OK

329                bad_1,

332                config,
333            )

347                lon_20_ku[i],

350                sat_vel_vec_20_ku[i],
351                inter_base_vec_20_ku[i],
352            )
353            final_lat_20_ku[i] = lat_poca
354            final_lon_20_ku[i] = lon_poca
355            height_20_ku[i] = elev_poca
356
357            if config["sin_geolocation"]["unwrap"]:
358                if phase > 0:
359                    unwrap_phase = -2.0 * np.pi + phase
360                else:
361                    unwrap_phase = 2.0 * np.pi + phase
362
363                angle_unwrap_20_ku[i] = phase_to_angle(unwrap_phase)
364                log.debug("-- GEOLOCATING  unwrapped --")
365                lat_poca, lon_poca, elev_poca = angle_to_poca(
366                    angle_unwrap_20_ku[i],
367                    lat_20_ku[i],
368                    lon_20_ku[i],
369                    alt_20_ku[i],
370                    range_cor_20_ku[i],
371                    sat_vel_vec_20_ku[i],
372                    inter_base_vec_20_ku[i],
373                )
374                lat_unwrap_20_ku[i] = lat_poca
375                lon_unwrap_20_ku[i] = lon_poca
376                height_unwrap_20_ku[i] = elev_poca

379                lon_unwrap_20_ku[i] = np.nan

381                delta_unwrap_h[i] = np.nan

385                or height_20_ku[i] < config["sin_geolocation"]["height_min"]
386            ):
387                raise sarin_phase.SINLocateError("Height out of bounds")
388        except OptimizeWarning as exc:
389            # Here to catch them for debugging but currently handled before here

391            # Doesn't indicate an error, indicates the model not fitting the data
392            raise exc
393        except sarin_phase.SINLocateError as exc:

396            final_lon_20_ku[i] = np.nan
397            height_20_ku[i] = np.nan

429        lon_initial_20_ku[:] = final_lon_20_ku[:]

431        # Use the alternate solution if better

439    log.info("Processed %d records", i + 1)
440
