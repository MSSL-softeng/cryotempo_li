 27def cnes_cycle_to_subcycle(cycle_number: int, rel_orbit_number: int) -> tuple[int, int]:
 28    """Calculate the CNES CS2 sub-cycle and track number number
 29
 30    Args:
 31        cycle_number (int): ESA CS2 cycle number from L1b
 32        rel_orbit_number (int): CS2 relative orbit within ESA cycle
 33
 34    Returns:
 35        tuple[int,int]: sub cycle number, track number
 36    """
 37    nb_tracks = 10688  # cycle of 368.24 days
 38    nb_sub_tracks = 840  # sub-cycle of 28.94 days
 39
 40    absolute_tr = (cycle_number - 1) * nb_tracks + rel_orbit_number
 41    sub_cy = ((absolute_tr - 1) // nb_sub_tracks) + 1
 42    sub_tr = ((absolute_tr - 1) % nb_sub_tracks) + 1
 43    return sub_cy, sub_tr

 46def get_current_commit_hash(log: logging.Logger) -> str:
 47    """retrieve the current git commit version
 48       or None if not available
 49
 50    Args:
 51        log (logging.Logger) : current log instance to use
 52
 53    Returns:
 54        str: git commit hash, or '' if failed
 55    """
 56    try:
 57        # Execute the Git command to get the commit hash
 58        result = subprocess.run(
 59            ["git", "rev-parse", "HEAD"], capture_output=True, text=True, check=True
 60        )
 61        commit_hash = result.stdout.strip()
 62        return commit_hash
 63    except subprocess.CalledProcessError as exc:
 64        # Handle the case when the command fails
 65        log.error("Could not get git commit hash, Error: %s", exc)
 66        return ""
