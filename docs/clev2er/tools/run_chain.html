clev2er.tools.run_chain

Main command line run control tool for CLEV2ER algorithm framework chains

Setup requires:

Set CLEV2ER_BASE_DIR to point to the base directory of the CLEV2ER framework export CLEV2ER_BASE_DIR=/Users/alanmuir/software/clev2er

PYTHONPATH to include $CLEV2ER_BASE_DIR/src export PYTHONPATH=$PYTHONPATH:$CLEV2ER_BASE_DIR/src

Example usage:

To list all command line options:

python run_chain.py -h

b) Run the cryotempo land ice chain on a single L2b file:

python run_chain.py --name cryotempo -f $CLEV2ER_BASE_DIR/testdata/cs2/l1bfiles/ CS_OFFL_SIR_LRM_1B_20200930T235609_20200930T235758_D001.nc

a) Run the cryotempo land ice chain on all l1b files in $CLEV2ER_BASE_DIR/testdata/cs2/l1bfiles

python run_chain.py --name cryotempo --dir $CLEV2ER_BASE_DIR/testdata/cs2/l1bfiles

Run with multi-processing and shared memory enabled (also can set these in main config):

python run_chain.py --name cryotempo -d $CLEV2ER_BASE_DIR/testdata/cs2/l1bfiles -sm -mp

  87def custom_key(line):
  88    """search function to find N in line containing [fN]
  89
  90    Args:
  91        line (str): string containing [fN], where N is an int which may be large
  92
  93    Returns:
  94        N or 0 if not matched
  95    """
  96    match = re.search(r"\[f(\d+)\]", line)
  97    if match:
  98        return int(match.group(1))
  99    return 0

 107    """

 117def insert_txtfile1_in_txtfile2_after_line_containing_string(
 118    file1: str, file2: str, target_string: str
 119) -> None:
 120    """Inserts txtfile1 in txtfile2 after line containing target_string
 121
 122    Args:
 123        file1 (str): path of txt file1
 124        file2 (str): path of txt file2
 125        target_string (str): string to search for in file2 and insert contents of file1 after
 126    """
 127    with open(file1, "r", encoding="utf-8") as fd1:
 128        content1 = fd1.read()
 129
 130    with open(file2, "r", encoding="utf-8") as fd2:
 131        lines = fd2.readlines()
 132
 133    new_lines = []
 134    for line in lines:
 135        new_lines.append(line)
 136        if target_string in line:
 137            new_lines.append(content1)
 138
 139    with open(file2, "w", encoding="utf-8") as fd2:
 140        fd2.writelines(new_lines)

 155def remove_strings_from_file(filename: str) -> None:
 156    """removes any string [fN] from the txt file
 157
 158        where N is any integer
 159
 160    Args:
 161        filename (str): file name
 162    """
 163    # Open the input file in read mode
 164    with open(filename, "r", encoding="utf-8") as file:
 165        # Read all the lines from the file
 166        lines = file.readlines()
 167
 168    # Create a regular expression pattern to match strings of the form '[fN]'
 169    pattern = r"\[f\d+\]"
 170
 171    # Remove the matched strings from each line
 172    modified_lines = [re.sub(pattern, "", line) for line in lines]
 173
 174    # Open the modified file in write mode
 175    with open(filename, "w", encoding="utf-8") as file:
 176        # Write the modified lines to the file
 177        file.writelines(modified_lines)

 180def run_chain_on_single_file(
 181    l1b_file: str,
 182    alg_object_list: list[Any],
 183    config: dict,
 184    log: logging.Logger,
 185    log_queue: Optional[Queue],
 186    rval_queue: Optional[Queue],
 187    filenum: int,
 188    breakpoint_alg_name: str = "",
 189) -> tuple[bool, str, str]:
 190    """Runs the algorithm chain on a single L1b file.
 191
 192       This function is run in a separate process if multi-processing is enabled.
 193
 194    Args:
 195        l1b_file (str): path of L1b file to process
 196        alg_object_list (list[Algorithm]): list of Algorithm objects
 197        log (logging.Logger): logging instance to use
 198        log_queue (Queue): Queue for multi-processing logging
 199        rval_queue (Queue) : Queue for multi-processing results
 200        filenum (int) : file number being processed
 201        breakpoint_alg_name (str) : if not '', name of algorithm to break after.
 202
 203    Returns:
 204        Tuple(bool,str,str):
 205        : algorithms success (True) or Failure (False),
 206        : '' or error string
 207        : path of breakpoint file or ''
 208        for multi-processing return values are instead queued -> rval_queue for this process
 209    """
 210
 211    bp_filename = ""  # break point file path
 212
 213    # Setup logging either for multi-processing or standard (single process)
 214    if config["chain"]["use_multi_processing"]:
 215        # create a logger
 216        logger = logging.getLogger("mp")
 217        # add a handler that uses the shared queue
 218        if log_queue is not None:
 219            handler = QueueHandler(log_queue)
 220            handler.setFormatter(logging.Formatter(f"[f{filenum}] %(message)s"))
 221            logger.addHandler(handler)
 222        # log all messages, debug and up
 223        logger.setLevel(logging.DEBUG)
 224        # get the current process
 225        process = current_process()
 226        # report initial message
 227        logger.debug("[f%d] Child %s starting.", filenum, process.name)
 228        thislog = logger
 229    else:
 230        thislog = log
 231
 232    thislog.info("_%s", "_" * 79)  # add a divider line in the log
 233
 234    thislog.info("Processing file %d: %s", filenum, l1b_file)
 235
 236    try:  # and open the NetCDF file
 237        with Dataset(l1b_file) as nc:
 238            # ------------------------------------------------------------------------
 239            # Run each algorithms .process() function in order
 240            # ------------------------------------------------------------------------
 241
 242            shared_dict = {"l1b_file_name": l1b_file}
 243
 244            for alg_obj in alg_object_list:
 245                alg_obj.set_filenum(filenum)
 246                alg_obj.set_log(thislog)
 247                # Run the Algorithm's process() function. Note that for multi-processing
 248                # the process() function also calls the init() function first
 249
 250                success, error_str = alg_obj.process(nc, shared_dict)
 251                if not success:
 252                    if "SKIP_OK" in error_str:
 253                        thislog.debug(
 254                            "Processing of L1b file %d : %s SKIPPED because %s",
 255                            filenum,
 256                            l1b_file,
 257                            error_str,
 258                        )
 259                    else:
 260                        thislog.error(
 261                            "Processing of L1b file %d : %s stopped because %s",
 262                            filenum,
 263                            l1b_file,
 264                            error_str,
 265                        )
 266
 267                    if config["chain"]["use_multi_processing"]:
 268                        if rval_queue is not None:
 269                            rval_queue.put((False, error_str, Timer.timers))
 270                        # Free up resources by running the Algorithm.finalize() on each
 271                        # algorithm instance
 272                        for alg_obj in alg_object_list:
 273                            if alg_obj.initialized:
 274                                alg_obj.finalize(stage=5)
 275                    return (False, error_str, bp_filename)
 276
 277                if alg_obj.alg_name.rsplit(".", maxsplit=1)[-1] == breakpoint_alg_name:
 278                    thislog.debug("breakpoint reached at algorithm %s", alg_obj.alg_name)
 279                    bp_filename = write_breakpoint_file(
 280                        config, shared_dict, thislog, breakpoint_alg_name
 281                    )
 282                    break
 283
 284            if config["chain"]["use_multi_processing"]:
 285                # Free up resources by running the Algorithm.finalize() on each
 286                # algorithm instance
 287                for alg_obj in alg_object_list:
 288                    if alg_obj.initialized:
 289                        alg_obj.finalize(stage=6)
 290
 291    except (IOError, ValueError, KeyError):
 292        error_str = f"Error processing {l1b_file}: {traceback.format_exc()}"
 293        thislog.error(error_str)
 294        if config["chain"]["use_multi_processing"]:
 295            if rval_queue is not None:
 296                rval_queue.put((False, error_str, Timer.timers))  # pass the function return values
 297                # back to the parent process
 298                # via a queue
 299        return (False, error_str, bp_filename)
 300
 301    if config["chain"]["use_multi_processing"]:
 302        if rval_queue is not None:
 303            rval_queue.put((True, "", Timer.timers))
 304    return (True, "", bp_filename)

 307def mp_logger_process(queue, config) -> None:
 308    """executed in a separate process that performs logging
 309       used for when multi-processing only
 310
 311    Args:
 312        queue (Queue): object created by multiprocessing.Queue()
 313        config (dict): main config dictionary for log file paths
 314    """
 315    # create a logger
 316    logger = logging.getLogger("mp")
 317    if config["log_files"]["debug"]:
 318        log_format = "[%(levelname)-2s] : %(asctime)s : %(name)-12s :  %(message)s"
 319    else:
 320        log_format = "[%(levelname)-2s] : %(asctime)s :  %(message)s"
 321
 322    log_formatter = logging.Formatter(log_format, datefmt="%d/%m/%Y %H:%M:%S")
 323
 324    # only includes ERROR level messages
 325    file_handler_error = logging.FileHandler(config["log_files"]["errors"] + ".mp", mode="w")
 326    file_handler_error.setFormatter(log_formatter)
 327    file_handler_error.setLevel(logging.ERROR)
 328    logger.addHandler(file_handler_error)
 329
 330    # include all allowed log levels up to INFO (ie ERROR, WARNING, INFO, not DEBUG)
 331    file_handler_info = logging.FileHandler(config["log_files"]["info"] + ".mp", mode="w")
 332    file_handler_info.setFormatter(log_formatter)
 333    file_handler_info.setLevel(logging.INFO)
 334    logger.addHandler(file_handler_info)
 335
 336    # include all allowed log levels up to DEBUG
 337    if config["log_files"]["debug"]:
 338        file_handler_debug = logging.FileHandler(config["log_files"]["debug"] + ".mp", mode="w")
 339        file_handler_debug.setFormatter(log_formatter)
 340        file_handler_debug.setLevel(logging.DEBUG)
 341        logger.addHandler(file_handler_debug)
 342
 343    # run forever
 344    while True:
 345        # consume a log message, block until one arrives
 346        message = queue.get()
 347        # check for shutdown
 348        if message is None:
 349            break
 350        # log the message
 351        logger.handle(message)

 354def run_chain(
 355    l1b_file_list: list[str],
 356    config: dict,
 357    algorithm_list: list[str],
 358    log: logging.Logger,
 359    breakpoint_alg_name: str = "",
 360) -> tuple[bool, int, int, int, str]:
 361    """Run the algorithm chain in algorithm_list on each L1b file in l1b_file_list
 362       using the configuration settings in config
 363
 364    Args:
 365        l1b_file_list (list[str]): list of l1b files paths to process
 366        config (dict): configuration dictionary. This is the named chain config and the
 367                                                 main config merged
 368        algorithm_list (list[str]): list of algorithm names
 369        log (logging.Logger): log instance to use
 370        breakpoint_alg_name (str): name of algorithm to set break point after.
 371                                   Default='' (no breakpoint set here)
 372
 373    Returns:
 374        tuple(bool,int,int, int,str) : (chain success or failure, number_of_errors,
 375                                  number of files processed, number of files skipped
 376                                  (for valid reasons), breakpoint filename)
 377    """
 378
 379    n_files = len(l1b_file_list)
 380    breakpoint_filename = ""
 381
 382    # -------------------------------------------------------------------------------------------
 383    # Load the dynamic algorithm modules from clev2er/algorithms/<algorithm_name>.py
 384    #   - runs each algorithm object's __init__() function
 385    # -------------------------------------------------------------------------------------------
 386    alg_object_list = []
 387    shared_mem_alg_object_list: List[Any] = []
 388    # duplicate list used to call initialization
 389    # of shared memory resources where used.
 390
 391    log.info("Dynamically importing and initializing algorithms from list...")
 392
 393    for alg in algorithm_list:
 394        # --------------------------------------------------------------------
 395        # Dynamically import each Algorithm from the list
 396        # --------------------------------------------------------------------
 397        try:
 398            module = importlib.import_module(
 399                f"clev2er.algorithms.{config['chain']['chain_name']}.{alg}"
 400            )
 401        except ImportError as exc:
 402            log.error("Could not import algorithm %s, %s", alg, exc)
 403            return (False, 1, 0, 0, breakpoint_filename)
 404
 405        # --------------------------------------------------------------------
 406        # Create an instance of each Algorithm,
 407        #   - runs its __init__(config) function
 408        # --------------------------------------------------------------------
 409
 410        # Load/Initialize algorithm
 411        try:
 412            alg_obj = module.Algorithm(config, log)
 413        except (FileNotFoundError, IOError, KeyError, ValueError):
 414            log.error("Could not initialize algorithm %s, %s", alg, traceback.format_exc())
 415            return (False, 1, 0, 0, breakpoint_filename)
 416
 417        alg_object_list.append(alg_obj)
 418
 419        # --------------------------------------------------------------------
 420        # Create a second instance of each Algorithm for multi-processing
 421        # shared memory buffer allocations,
 422        #   - runs its __init__(config) function
 423        # Note that the .process() function is never run for this instance
 424        # We merge  {"_init_shared_mem": True} to the config so that the
 425        # Algorithm knows to run any shared memory initialization
 426        # --------------------------------------------------------------------
 427
 428        if config["chain"]["use_multi_processing"] and config["chain"]["use_shared_memory"]:
 429            # Load/Initialize algorithm
 430            try:
 431                alg_obj_shm = module.Algorithm(config | {"_init_shared_mem": True}, log)
 432            except (FileNotFoundError, IOError, KeyError, ValueError):
 433                log.error(
 434                    "Could not initialize algorithm for shared_memory %s, %s",
 435                    alg,
 436                    traceback.format_exc(),
 437                )
 438                # If there is a failure we must clean up any shared memory already allocated
 439                for alg_obj_shm in shared_mem_alg_object_list:
 440                    if alg_obj_shm.initialized:
 441                        alg_obj_shm.finalize(stage=4)
 442
 443                return (False, 1, 0, 0, breakpoint_filename)
 444
 445            shared_mem_alg_object_list.append(alg_obj_shm)
 446
 447        # If a breakpoint after this alg is set we don't need to initialize any more algorithms
 448        if alg == breakpoint_alg_name:
 449            log.debug("breakpoint reached in import for %s", alg)
 450            break
 451    # -------------------------------------------------------------------------------------------
 452    #  Run algorithm chain's Algorthim.process() on each L1b file in l1b_file_list
 453    #    - a different method required for multi-processing or standard sequential processing
 454    #    - Note that choice of MP method is due to logging reliability constraints, which
 455    #      caused problems with simpler more modern pool.starmap methods
 456    # -------------------------------------------------------------------------------------------
 457    num_errors = 0
 458    num_files_processed = 0
 459    num_skipped = 0
 460
 461    # --------------------------------------------------------------------------------------------
 462    # Parallel Processing (optional)
 463    # --------------------------------------------------------------------------------------------
 464    if config["chain"]["use_multi_processing"]:  # pylint: disable=R1702
 465        # With multi-processing we need to redirect logging to a stream
 466
 467        # create a shared logging queue for multiple processes to use
 468        log_queue: Queue = Queue()
 469
 470        # create a logger
 471        new_logger = logging.getLogger("mp")
 472        # add a handler that uses the shared queue
 473        new_logger.addHandler(QueueHandler(log_queue))
 474        # log all messages, debug and up
 475        new_logger.setLevel(log.level)
 476        # start the logger process
 477        logger_p = Process(target=mp_logger_process, args=(log_queue, config))
 478        logger_p.start()
 479
 480        # Divide up the input files in to chunks equal to maximum number of processes
 481        # allowed == config["chain"]["max_processes_for_multiprocessing"]
 482
 483        log.info(
 484            "Using multi-processing with max %d processes",
 485            config["chain"]["max_processes_for_multiprocessing"],
 486        )
 487
 488        num_chunks = ceil(n_files / config["chain"]["max_processes_for_multiprocessing"])
 489
 490        file_indices_chunks = np.array_split(list(range(n_files)), num_chunks)
 491
 492        for chunk_num, file_indices in enumerate(file_indices_chunks):
 493            chunked_l1b_file_list = np.array(l1b_file_list)[file_indices]
 494            log.debug(f"mp chunk_num {chunk_num}: chunked_l1b_file_list={chunked_l1b_file_list}")
 495
 496            num_procs = len(chunked_l1b_file_list)
 497
 498            log.info(
 499                "Running process set %d of %d (containing %d processes)",
 500                chunk_num + 1,
 501                num_chunks,
 502                num_procs,
 503            )
 504
 505            # Create separate queue for each new process to handle function return values
 506            rval_queues: List[Queue] = [Queue() for _ in range(num_procs)]
 507
 508            # configure child processes
 509            processes = [
 510                Process(
 511                    target=run_chain_on_single_file,
 512                    args=(
 513                        chunked_l1b_file_list[i],
 514                        alg_object_list,
 515                        config,
 516                        None,
 517                        log_queue,
 518                        rval_queues[i],
 519                        file_indices[i],
 520                        breakpoint_alg_name,
 521                    ),
 522                )
 523                for i in range(num_procs)
 524            ]
 525            # start child processes
 526            for process in processes:
 527                process.start()
 528
 529            # wait for child processes to finish
 530            for i, process in enumerate(processes):
 531                process.join()
 532                # retrieve return values of each process function from queue
 533                # rval=(bool, str, Timer.timers)
 534                while not rval_queues[i].empty():
 535                    rval = rval_queues[i].get()
 536                    if not rval[0] and "SKIP_OK" not in rval[1]:
 537                        num_errors += 1
 538                    if "SKIP_OK" in rval[1]:
 539                        num_skipped += 1
 540                    num_files_processed += 1
 541                    # rval[2] returns the Timer.timers dict for algorithms process() function
 542                    # ie a dict containing timers['alg_name']= the number of seconds elapsed
 543                    for key, value in rval[2].items():
 544                        if key in Timer.timers:
 545                            Timer.timers.add(key, value)
 546                        else:
 547                            Timer.timers.add(key, value)
 548
 549        # shutdown the queue correctly
 550        log_queue.put(None)
 551
 552        log.info("MP processing completed with outputs logged:")
 553    # --------------------------------------------------------------------------------------------
 554    # Sequential Processing
 555    # --------------------------------------------------------------------------------------------
 556    else:  # Normal sequential processing (when multi-processing is disabled)
 557        try:
 558            for fnum, l1b_file in enumerate(l1b_file_list):
 559                log.info("\n%sProcessing file %d of %d%s", "-" * 20, fnum, n_files, "-" * 20)
 560                success, error_str, breakpoint_filename = run_chain_on_single_file(
 561                    l1b_file,
 562                    alg_object_list,
 563                    config,
 564                    log,
 565                    None,
 566                    None,
 567                    fnum,
 568                    breakpoint_alg_name,
 569                )
 570                num_files_processed += 1
 571                if not success and "SKIP_OK" in error_str:
 572                    log.debug("Skipping file")
 573                    num_skipped += 1
 574                    continue
 575                if not success:
 576                    num_errors += 1
 577
 578                    if config["chain"]["stop_on_error"]:
 579                        log.error(
 580                            "Chain stopped because of error processing L1b file %s",
 581                            l1b_file,
 582                        )
 583                        break
 584
 585                    log.error(
 586                        "Error processing L1b file %s, skipping file",
 587                        l1b_file,
 588                    )
 589                    continue
 590        except KeyboardInterrupt as exc:
 591            log.error("KeyboardInterrupt detected", exc)
 592
 593    # -----------------------------------------------------------------------------
 594    # Run each algorithms .finalize() function in order
 595    # -----------------------------------------------------------------------------
 596
 597    log.debug("_" * 79)  # add a divider line in the log
 598
 599    for alg_obj_shm in shared_mem_alg_object_list:
 600        if alg_obj_shm.initialized:
 601            alg_obj_shm.finalize(stage=2)
 602
 603    for alg_obj in alg_object_list:
 604        if alg_obj.initialized:
 605            alg_obj.finalize(stage=3)
 606
 607    # Elapsed time for each algorithm.
 608    # Note if multi-processing, process times are added for each algorithm
 609    # (so total time processing will be less)
 610    #  - ie for processes p1 and p2,
 611    #  -    alg1.time =(p1.alg1.time +p2.alg1.time +,..)
 612    #  -    alg2.time =(p1.alg2.time +p2.alg2.time +,..)
 613
 614    log.info("\n%sAlgorithm Cumulative Processing Time%s", "-" * 20, "-" * 20)
 615
